## Usage Examples

### Navigate to Specific Module
```tsx
import { router } from 'expo-router';

// Navigate to 3D Modeling
router.push('/modules/3d-modeling');

// Navigate to Budget Management
router.push('/modules/budget-management');
```

### Filter Modules by Category
```tsx
import { CATEGORIES } from '@/constants/categories';

const category = CATEGORIES.find(c => c.id === 'architecture');
const modules = category?.modules || [];

// Get all architecture modules
modules.forEach(module => {
  console.log(module.label, module.route);
});
```

### Search Modules
```tsx
import { CATEGORIES } from '@/constants/categories';

function searchModules(query: string) {
  const results = [];
  
  CATEGORIES.forEach(category => {
    category.modules.forEach(module => {
      if (module.label.toLowerCase().includes(query.toLowerCase())) {
        results.push({
          ...module,
          categoryName: category.label,
          categoryColor: category.color,
        });
      }
    });
  });
  
  return results;
}

// Usage
const results = searchModules('thiết kế');
```
